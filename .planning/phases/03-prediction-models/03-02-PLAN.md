```python
from dataclasses import dataclass
from abc import ABC, abstractmethod

@dataclass
class PredictionResult:
    value: float
    confidence_lower: float
    confidence_upper: float
    metadata: dict

class BasePredictor(ABC):
    @abstractmethod
    def train(self, features: list[dict], targets: list[float]) -> None: ...
    @abstractmethod
    def predict(self, features: dict) -> PredictionResult: ...
    @abstractmethod
    def get_feature_names(self) -> list[str]: ...
    def save(self, path: str) -> None: ...
    @classmethod
    def load(cls, path: str) -> "BasePredictor": ...
    def cross_validate(self, features, targets, cv=5) -> dict: ...
    @staticmethod
    def features_to_array(features: dict, feature_names: list[str]) -> list[float]: ...
```

       - `GAME_FEATURE_COLS`: The TeamFeatures columns to use: ["pace", "offensive_rating", "defensive_rating", "rest_days", "season_win_pct"]. For each, generate home_X, away_X, diff_X (home minus away). Plus `is_home = 1.0` as explicit home court feature. Total: 5*3 + 1 = 16 features.

       - Handle NaN/None in features: fill with 0.0 via features_to_array
